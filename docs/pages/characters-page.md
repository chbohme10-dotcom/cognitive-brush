```sql
CREATE TABLE character_assets (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    character_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    asset_type VARCHAR(50), -- 'image', 'voice', 'video'
    asset_url TEXT NOT NULL,
    metadata JSONB,
    quality_score FLOAT,
    ai_model VARCHAR(100), -- Which AI model generated this
    generation_params JSONB, -- Parameters used for generation
    created_at TIMESTAMP DEFAULT NOW()
);

-- Character Relationships
CREATE TABLE character_relationships (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    character_a_id UUID REFERENCES characters(id),
    character_b_id UUID REFERENCES characters(id),
    relationship_type VARCHAR(100),
    relationship_strength FLOAT DEFAULT 0.0,
    interaction_history JSONB,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Enable Row Level Security
ALTER TABLE characters ENABLE ROW LEVEL SECURITY;
ALTER TABLE character_assets ENABLE ROW LEVEL SECURITY;
ALTER TABLE character_relationships ENABLE ROW LEVEL SECURITY;

-- RLS Policies
CREATE POLICY "Users can manage their own characters" ON characters
    FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "Users can manage their character assets" ON character_assets
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM characters 
            WHERE characters.id = character_assets.character_id 
            AND characters.user_id = auth.uid()
        )
    );
```
